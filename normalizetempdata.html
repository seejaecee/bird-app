import pandas as pd
import worldbuilder as wb
contenents = {"northamerica":1, "southamerica":2, "europe":3, "africa":4, "asia":5, "oceania":6}
import os
try: 
    os.mkdir("tempdata")
    print("making file directory for temp data....")
except:
    pass

tm = pd.read_excel("rawtempdata/annualtemperaturedata.ods", engine="odf")
countries = ["China", "France", "Germany", "India", "United States of America",
                         "United Kingdom"]
#latitutdes and longitudes of each continent as defined in the documentation
#for da NOAA datasets. 
#blank (4 now)dataframe to add data to thru looping

cntlats = {6: [54.75, -9.24], 5: [-12.20, 81.72], 4:[-46.97,35.93],
           3: [-54.46, 80.77], 2: [-55.92, 15.70], 1:[-0.39, 83.17]}
cntlongs = {6: [73.24, 168.00], 5: [-180.00,180.00], 
            4: [-25.36,51.42], 3: [-68.42, 55.85], 2: [-109.45,-28.88], 1:[-179.14, 179.78]}
allcols = ["Year", "Month", "MonthlyAnomaly", "MAUnc", "AnnualAnomaly", 
             "AAUnc", "FiveYrAnomaly", "FYUnc", "TenYrAnomaly", 
             "TYUnc", "TwentyYrAnomaly", "TWYUnc", "Contenent", "Lat", "Long"] 
rg = pd.DataFrame(columns=allcols) #rg= REGIONAL

for cnt in contenents.keys():
    df = pd.read_excel(f"rawtempdata/regional/{cnt}.ods", skiprows=1, header=None,engine="odf") 
    df.columns = allcols[0:12]
    contcode = contenents[cnt]
    df[allcols[12]] = [contcode]*len(df)
    df[allcols[13]] = [cntlats[contcode]]*len(df)
    df[allcols[14]] = [cntlongs[contcode]]*len(df)
    rg = pd.concat([rg, df], ignore_index=True)
#this is a bit fucky but. NOAA randomly had all their datasets as .txt not .csv
#files. which was evil........... anyway copy pasting the data hella fucked up 
#the column names and i didnt wanna change them all individually in excel
#bc i was lazy. so. we just drop that first row and replayce w/the actual correct names
rg.to_csv("tempdata/regional_tempdata.csv", sep='\t', encoding='utf-8')

ct = pd.DataFrame()
for cnt in countries:
    df = pd.read_excel(f"rawtempdata/country/{cnt.lower()}.ods", skiprows=1, header=None, engine="odf") 
    df.columns = allcols[0:12]
    df["Country"] = [cnt]*len(df)
    ct = pd.concat([ct, df], ignore_index=True)
ct.to_csv("tempdata/country_tempdata.csv", sep='\t', encoding='utf-8')
anncols = ["Year", "Annual Anomaly", "Annual Unc.", "Five-year Anomaly", "Five-year Unc."]
annual = pd.DataFrame(columns=["Year", "Annual Anomaly", "Annual Unc.", "Five-year Anomaly", "Five-year Unc."])

annual["Year"]= tm.iloc[:,0]
for c1 in range(1,5):
    annual[anncols[c1]] = 0.5*(tm.iloc[:,c1] + tm.iloc[:,c1+4])
annual.to_csv("tempdata/annual_tempdata.csv", sep='\t', encoding='utf-8')
