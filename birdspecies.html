def common_name(sp):
    """gets the common name of a species by the ebird database"""
    try: 
        return birdnames[birdnames['SCIENTIFIC_NAME'] == sp]['COMMON_NAME'].values[0]
    except: 
        return sp

def download_blurb(sp):
    """downloads a blurb about the species"""
    cn = common_name(species)    
    url = f"https://en.wikipedia.org/api/rest_v1/page/summary/{cn}"
    #so i basically need to put identifying header on my request otherwise wikipedia 
    #will deny it bc it doesn't allow bots. 
    headers = {
        "User-Agent": "BirdDB/1.0 (https://mines.edu; contact: cjcarpenter@mines.edu)"
    }
    reqest = requests.get(url, headers=headers, timeout=10)
    if reqest.status_code == 404:
        return None
    
    data = reqest.json()
    return data.get("extract")
