import geopandas as gpd
url = "https://d2ad6b4ur7yvpq.cloudfront.net/naturalearth-3.3.0/ne_110m_land.geojson"
world = gpd.read_file(url)
contcodes = {"AF":"africa", 
             "AS":"asia",
             "EU":"europe",
             "NA":"northamerica",
             "OC":"oceania",
             "SA":"southamerica"}
conts = {}
for cnt in contcodes.keys(): 
    conts[contcodes[cnt]] = gpd.read_file(f"json/{contcodes[cnt]}.geojson")
