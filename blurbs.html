"""
title: BLURBS.PY
author: CJ CARPENTER
description: makes a csv file for species & blurb"""
import birdspecies as bs
import pandas as pd
blurbs = {"name": [], "blurb": []}
for sp in bs.species: 
    blurbs["name"].append(sp)
    try: 
        blurbs["blurb"].append(bs.download_blurb(sp))
    except: 
        blurbs["blurb"].append("no info")
data = pd.DataFrame.from_dict(blurbs)
data.to_csv("blurbs.csv", sep='\t', encoding='utf-8')
