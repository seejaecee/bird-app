def send_plumes(queries):
    """goes through the dataset to find unique bird species then searches google and gets pictures of them and 
    then downloads them yay"""
    #okkkkkkkk so we first get the unique bird species in the dataset. 
    
    for bird in queries: 
        #now we r adding it to a url um. idk. i just copied all the text from the google images search query 
        #and inserted it in there lol 
        url = f"https://www.google.com/search?q={bird}&source=lnms&tbm=isch&sa=X&ved=2ahUKEwie44_AnqHpAhUhBWMBHUFGD90Q_AUoAXoECBUQAw&biw=1920&bih=947"
        response = requests.get(url)
        if response.status_code != 200: 
            #so this would be an error code if our link stopped working. idk.
            #i didnt want to do RaiseException because uhhhh 
            #exceptions stress me out lol
            print(f"couldn't find {bird} idk moving onto next")
            continue
        #we get all the html on the page basically
        soup = BeautifulSoup(response.text, 'html.parser')
        #i was scouring thru the raw html in inspectelement and figured out that 
        #everything that starts w src= is an image 
        #so we will grab all those
        soup_images = re.findall(rf'src="(\S+)"', str(soup))
        birds_left = [soup_images]
        i = 0 
        #so this is a kinda convoluted way 2 do it but the native google images on the page don't start
        #w/ https:/ instead they start 
        #with something like \image so this is meant to exclude those and just get the proper bird pics
        while soup_images[i][0:4] != 'http':
            birds_left = soup_images[i+1::]
            i += 1 
        bird_found = requests.get(birds_left[0]).content
        #downloading image (NOTE 2 SELF MUST USE FORWARD SLASH IN DIR FOR LINUX!!!!!!!!!!!)
        img = open(f"bird_pics/{bird}.jpg", 'wb')
        img.write(bird_found)
        img.close()
